// src/pages/GroupViewPage.tsx
import React, { useState, useEffect, useRef, useCallback } from "react";
import { Link } from "react-router-dom";
import ImageSequencePlayer from "../components/ImageSequencePlayer";

// --- Type Definitions ---
interface Track {
  track_id: number;
  global_id?: string | null;
  bbox_xyxy: [number, number, number, number];
  confidence?: number | null;
  class_id?: number | null;
  map_coords?: [number, number] | null;
}

interface CameraDataFromWS {
  image_source: string;
  tracks: Track[];
}

interface WebSocketTrackingPayload {
  frame_index: number;
  scene_id: string;
  timestamp_processed_utc: string;
  cameras: {
    [jsonCameraId: string]: CameraDataFromWS;
  };
}

interface TaskStatusPayload {
    task_id?: string;
    status: string;
    progress: number;
    current_step?: string | null;
    details?: string | null;
    start_time?: string;
    last_update_time?: string;
    environment_id?: string;
}

interface CameraConfig {
  basePath: string;
  startFrame: number;
  frameCount: number;
  extension?: string;
}

// --- Configuration & Constants ---
const zoneName = "Factory";
const appCameraIds = ["camera1", "camera2", "camera3", "camera4"];
const cameraNames = ["Camera 1", "Camera 2", "Camera 3", "Camera 4"];

const cameraFrameConfig: { [appId: string]: CameraConfig } = {
  camera1: { basePath: "/frames/camera1/", startFrame: 0, frameCount: 51, extension: "jpg" },
  camera2: { basePath: "/frames/camera2/", startFrame: 0, frameCount: 51, extension: "jpg" },
  camera3: { basePath: "/frames/camera3/", startFrame: 0, frameCount: 51, extension: "jpg" },
  camera4: { basePath: "/frames/camera4/", startFrame: 0, frameCount: 51, extension: "jpg" }
};

const appCameraIdToJsonId: { [appId: string]: string } = {
    'camera1': 'c09',
    'camera2': 'c12',
    'camera3': 'c13',
    'camera4': 'c16',
};
// eslint-disable-next-line @typescript-eslint/no-unused-vars
const jsonIdToAppCameraId: { [jsonId: string]: string } = Object.fromEntries(
  Object.entries(appCameraIdToJsonId).map(([appId, jsonId]) => [jsonId, appId])
);

const MAP_SOURCE_WIDTH = 1920;
const MAP_SOURCE_HEIGHT = 1080;

interface SingleCameraMapPoint {
    x: number;
    y: number;
    globalId: string;
}

const cameraColorMap: { [jsonCameraId: string]: string } = {
    'c09': 'bg-cyan-400',
    'c12': 'bg-red-500',
    'c13': 'bg-yellow-400',
    'c16': 'bg-purple-500',
};
const defaultDotColor = 'bg-gray-500';

const BACKEND_HTTP_URL = "http://localhost:8000";
const BACKEND_WS_URL_BASE = "ws://localhost:8000";
const DEFAULT_ENVIRONMENT_ID = "factory";

type TabType = "all" | string;

// --- GroupViewPage Component ---
const GroupViewPage: React.FC = () => {
  const [activeTab, setActiveTab] = useState<TabType>("all");
  const [taskId, setTaskId] = useState<string | null>(null);
  const [taskStatus, setTaskStatus] = useState<TaskStatusPayload | null>(null);
  const [websocketState, setWebsocketState] = useState<WebSocket | null>(null);
  const [isConnecting, setIsConnecting] = useState<boolean>(false);
  const [connectionError, setConnectionError] = useState<string | null>(null);
  const [currentFrameData, setCurrentFrameData] = useState<WebSocketTrackingPayload | null>(null);
  const [currentGlobalFrameIndex, setCurrentGlobalFrameIndex] = useState<number>(0);

  const overallMapContainerRef = useRef<HTMLDivElement>(null);
  const [overallMapDimensions, setOverallMapDimensions] = useState({ width: 0, height: 0 });
  const [perCameraMapPoints, setPerCameraMapPoints] = useState<{ [jsonCameraId: string]: SingleCameraMapPoint[] }>({});
  
  const wsInstanceRef = useRef<WebSocket | null>(null);
  // No longer using initialLoadEffectRan, relying on startTaskAndConnect's idempotency for cleanup

  const startTaskAndConnect = useCallback(async (environmentId: string, isNewTaskRequest: boolean = true) => {
    if (isConnecting) {
      console.log("startTaskAndConnect: IGNORED - already connecting.");
      return;
    }
    console.log(`startTaskAndConnect: CALLED with env: ${environmentId}, isNewTaskRequest: ${isNewTaskRequest}, current taskId state: ${taskId}`);

    const previousWsInstance = wsInstanceRef.current;
    if (previousWsInstance) {
        console.log("startTaskAndConnect: Cleaning up previous WebSocket instance.");
        previousWsInstance.onopen = null;
        previousWsInstance.onmessage = null;
        previousWsInstance.onerror = null;
        previousWsInstance.onclose = null;
        if (previousWsInstance.readyState === WebSocket.OPEN || previousWsInstance.readyState === WebSocket.CONNECTING) {
            previousWsInstance.close(1000, "Superseded by new connection attempt");
            console.log("startTaskAndConnect: Called close() on previous WebSocket instance.");
        }
        wsInstanceRef.current = null;
    }
    setWebsocketState(null);

    setIsConnecting(true);
    setConnectionError(null);
    
    let currentProcessingTaskId = taskId; 
    if (isNewTaskRequest || !currentProcessingTaskId) {
        console.log("startTaskAndConnect: Will start a NEW task (or taskId was null). Resetting related states.");
        setCurrentFrameData(null); setTaskStatus(null);
        setPerCameraMapPoints({}); setTaskId(null);
        currentProcessingTaskId = null;
    } else {
        console.log(`startTaskAndConnect: Will attempt to use existing taskId: ${currentProcessingTaskId}`);
    }

    let tempWs: WebSocket | null = null;

    try {
      let wsPathFromResponse: string | null = null;
      if (!currentProcessingTaskId) { 
        console.log(`startTaskAndConnect: Requesting to start NEW processing task for environment: ${environmentId}`);
        const response = await fetch(`${BACKEND_HTTP_URL}/api/v1/processing-tasks/start`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ environment_id: environmentId }),
        });
        if (!response.ok) {
          const errorText = await response.text().catch(() => `HTTP error ${response.status}`);
          throw new Error(`Failed to start task: ${response.status} - ${errorText}`);
        }
        const data = await response.json();
        currentProcessingTaskId = data.task_id;
        wsPathFromResponse = data.websocket_url;
        if (!currentProcessingTaskId || !wsPathFromResponse) throw new Error("Task ID or WebSocket URL missing from new task response.");
        setTaskId(currentProcessingTaskId);
      } else {
        wsPathFromResponse = `/ws/tracking/${currentProcessingTaskId}`;
        console.log(`startTaskAndConnect: Attempting to connect/reconnect to existing task: ${currentProcessingTaskId}`);
      }

      const wsUrl = `${BACKEND_WS_URL_BASE}${wsPathFromResponse!.startsWith('/') ? wsPathFromResponse : '/' + wsPathFromResponse}`;
      console.log(`startTaskAndConnect: Attempting to new WebSocket(${wsUrl})`);
      
      tempWs = new WebSocket(wsUrl);
      wsInstanceRef.current = tempWs; // Update ref immediately to the new instance

      tempWs.onopen = () => {
        if (wsInstanceRef.current === tempWs) { 
            console.log(`WebSocket connected for task: ${currentProcessingTaskId} to ${wsUrl} (instance in ref MATCHES)`);
            setIsConnecting(false);
            setWebsocketState(tempWs);
            setConnectionError(null);
        } else {
            console.warn(`WebSocket opened for an outdated attempt (task ${currentProcessingTaskId}). wsInstanceRef.current is different. Closing this (tempWs).`);
            tempWs?.close(1000, "Outdated WebSocket attempt onopen");
        }
      };

      tempWs.onmessage = (event) => {
        if (wsInstanceRef.current !== tempWs) {
            // console.log("Ignoring message on outdated WebSocket instance.");
            return;
        }
        try {
          const message = JSON.parse(event.data as string);
          console.log("WebSocket Message Received (current instance):", JSON.stringify(message, null, 2)); 
          if (message.type === "tracking_update") {
            const payload = message.payload as WebSocketTrackingPayload;
            console.log("Tracking Update Payload (current instance):", JSON.stringify(payload, null, 2));
            setCurrentFrameData(payload);
            setCurrentGlobalFrameIndex(payload.frame_index);
          } else if (message.type === "status_update") {
            console.log("Status Update Payload (current instance):", JSON.stringify(message.payload, null, 2));
            setTaskStatus(message.payload as TaskStatusPayload);
          }
        } catch (error) {
          console.error("Error processing WebSocket message:", error);
        }
      };

      tempWs.onerror = (event) => {
        console.error("WebSocket error event on tempWs:", event);
        if (wsInstanceRef.current === tempWs) {
            setConnectionError("WebSocket connection error.");
            setIsConnecting(false);
            wsInstanceRef.current = null; 
            setWebsocketState(null);
        } else {
            console.warn("Error on an outdated WebSocket instance (tempWs), wsInstanceRef.current is different or null.");
        }
      };

      tempWs.onclose = (event) => {
        console.log(`WebSocket (tempWs instance) closed. Task: ${taskId || currentProcessingTaskId}, Code: ${event.code}, Reason: '${event.reason}', Clean: ${event.wasClean}`);
        if (wsInstanceRef.current === tempWs) { // If this was the active WebSocket referenced by wsInstanceRef
            setIsConnecting(false);
            wsInstanceRef.current = null; // Clear the ref
            setWebsocketState(null); // Clear the state
            if (!event.wasClean) {
                setConnectionError(`WebSocket closed unexpectedly (Code: ${event.code}).`);
            }
        } else {
            console.warn("onclose event for an outdated/already cleaned WebSocket instance (tempWs). wsInstanceRef.current is different or null.");
        }
      };
    } catch (error) {
      console.error("Error in startTaskAndConnect outer try-catch:", error);
      setConnectionError((error as Error).message || "Failed to start task or connect.");
      setIsConnecting(false);
      
      const wsToCloseOnError = wsInstanceRef.current === tempWs ? wsInstanceRef.current : tempWs;
      if (wsToCloseOnError) {
        console.log("Closing WebSocket due to error in startTaskAndConnect (outer catch).");
        wsToCloseOnError.onopen = null; wsToCloseOnError.onmessage = null;
        wsToCloseOnError.onerror = null; wsToCloseOnError.onclose = null;
        if (wsToCloseOnError.readyState === WebSocket.OPEN || wsToCloseOnError.readyState === WebSocket.CONNECTING) {
            wsToCloseOnError.close(1000, "Error during connection setup");
        }
      }
      if (wsInstanceRef.current === wsToCloseOnError) wsInstanceRef.current = null;
      setWebsocketState(null);
    }
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [isConnecting, taskId]);

  useEffect(() => {
    // This effect will run on mount. In StrictMode (dev), it runs, cleans up, then runs again.
    // The `startTaskAndConnect` function is designed to handle being called again by cleaning
    // up any previous WebSocket instance stored in `wsInstanceRef.current`.
    console.log("Mount useEffect: Calling startTaskAndConnect.");
    startTaskAndConnect(DEFAULT_ENVIRONMENT_ID, true); // true for isNewTask

    return () => {
      console.log("Mount useEffect: Cleanup function running.");
      const wsToCleanup = wsInstanceRef.current;
      if (wsToCleanup) {
        console.log("Mount useEffect Cleanup: Closing WebSocket instance via ref.");
        // Detach handlers first to prevent them firing during/after close
        wsToCleanup.onopen = null;
        wsToCleanup.onmessage = null;
        wsToCleanup.onerror = null;
        wsToCleanup.onclose = null; 
        if (wsToCleanup.readyState === WebSocket.OPEN || wsToCleanup.readyState === WebSocket.CONNECTING) {
            wsToCleanup.close(1000, "Component unmounting or effect re-running");
        }
        wsInstanceRef.current = null; // Clear the ref
      }
      setWebsocketState(null); // Also clear state
    };
  // eslint-disable-next-line react-hooks/exhaustive-deps
  }, [startTaskAndConnect]); // Re-run if startTaskAndConnect identity changes.

  useEffect(() => {
    const mapElement = overallMapContainerRef.current;
    if (!mapElement) return;
    const resizeObserver = new ResizeObserver(entries => {
      for (let entry of entries) { setOverallMapDimensions({ width: entry.contentRect.width, height: entry.contentRect.height }); }
    });
    resizeObserver.observe(mapElement);
    setOverallMapDimensions({ width: mapElement.offsetWidth, height: mapElement.offsetHeight });
    return () => resizeObserver.disconnect();
  }, []);

  useEffect(() => {
    if (!currentFrameData || !overallMapDimensions.width || !overallMapDimensions.height || MAP_SOURCE_WIDTH <= 0 || MAP_SOURCE_HEIGHT <= 0) {
      setPerCameraMapPoints({}); return;
    }
    const newPerCameraPoints: { [jsonCameraId: string]: SingleCameraMapPoint[] } = {};
    const quadrantWidth = overallMapDimensions.width / 2;
    const quadrantHeight = overallMapDimensions.height / 2;
    const scaleX = quadrantWidth / MAP_SOURCE_WIDTH;
    const scaleY = quadrantHeight / MAP_SOURCE_HEIGHT;

    Object.entries(currentFrameData.cameras).forEach(([jsonCameraId, cameraData]) => {
      newPerCameraPoints[jsonCameraId] = [];
      if (cameraData?.tracks) {
        cameraData.tracks.forEach(track => {
          if (track.map_coords && typeof track.global_id === 'string') {
            const [mapX, mapY] = track.map_coords;
            newPerCameraPoints[jsonCameraId].push({
              x: mapX * scaleX, y: mapY * scaleY, globalId: track.global_id
            });
          }
        });
      }
    });
    setPerCameraMapPoints(newPerCameraPoints);
  }, [currentFrameData, overallMapDimensions]);

  const handleReconnectOrNewTask = () => {
    console.log("handleReconnectOrNewTask called: Forcing new task.");
    const wsToClose = wsInstanceRef.current;
    if (wsToClose) {
        wsToClose.onopen = null; wsToClose.onmessage = null;
        wsToClose.onerror = null; wsToClose.onclose = null;
        if (wsToClose.readyState === WebSocket.OPEN || wsToClose.readyState === WebSocket.CONNECTING) {
            wsToClose.close(1000, "User initiated new task/reconnect");
        }
        wsInstanceRef.current = null;
    }
    setWebsocketState(null);
    setTaskId(null); 
    startTaskAndConnect(DEFAULT_ENVIRONMENT_ID, true);
  };

  const activeCameraCount = currentFrameData ? Object.values(currentFrameData.cameras).filter(cam => cam.tracks && cam.tracks.length > 0).length : 0;
  const inactiveCameraCount = appCameraIds.length - activeCameraCount;

  return (
    <div className="flex flex-col h-screen bg-gray-900 text-gray-200 p-4 sm:p-6">
      <header className="flex items-center justify-between mb-4 flex-shrink-0">
        <Link to="/" className="flex items-center text-lg hover:text-orange-400">
            <svg xmlns="http://www.w3.org/2000/svg" className="h-5 w-5 mr-2" viewBox="0 0 20 20" fill="currentColor"><path fillRule="evenodd" d="M12.707 5.293a1 1 0 010 1.414L9.414 10l3.293 3.293a1 1 0 01-1.414 1.414l-4-4a1 1 0 010-1.414l4-4a1 1 0 011.414 0z" clipRule="evenodd" /></svg>
            Back
        </Link>
        <h1 className="text-2xl font-semibold">{currentFrameData?.scene_id || zoneName}</h1>
        <div className="text-xs text-right space-y-0.5 w-1/4 sm:w-1/3 md:w-1/4 lg:w-1/5">
            {taskId && <div>Task: <span className="font-mono">{taskId.substring(0,8)}...</span></div>}
            {taskStatus && <div>Status: {taskStatus.status} ({(taskStatus.progress * 100).toFixed(0)}%)</div>}
            {taskStatus?.current_step && <div className="truncate" title={taskStatus.current_step}>{taskStatus.current_step}</div>}
            {connectionError && <div className="text-red-400 font-semibold">{connectionError}</div>}
        </div>
      </header>

      <div className="flex flex-col sm:flex-row justify-between items-start sm:items-center mb-4 bg-gray-800 p-3 rounded-md flex-shrink-0 gap-4">
        <div className="flex flex-col sm:flex-row sm:items-center gap-x-4 gap-y-1">
            <div>Num cameras: <span className="font-semibold">{appCameraIds.length}</span></div>
            <div className="flex space-x-4">
                <span className="text-green-400 flex items-center"><span className="h-2 w-2 bg-green-400 rounded-full mr-1"></span> Active: {activeCameraCount}</span>
                <span className="text-red-400 flex items-center"><span className="h-2 w-2 bg-red-400 rounded-full mr-1"></span> Inactive: {inactiveCameraCount}</span>
            </div>
            <div>Map: <span className="font-semibold">floor 1</span></div>
        </div>
        <div className="flex space-x-2 flex-shrink-0">
            <button
                onClick={handleReconnectOrNewTask}
                disabled={isConnecting}
                className={`px-4 py-1.5 rounded text-white text-sm font-semibold ${ isConnecting ? "bg-gray-600 cursor-not-allowed" : (websocketState ? "bg-green-600 hover:bg-green-700" : "bg-blue-600 hover:bg-blue-700") }`}
            >
                {isConnecting ? "Connecting..." : (websocketState ? "Connected (Start New Task)" : "Connect / Start Task")}
            </button>
        </div>
      </div>

      <div className="mb-4 border-b border-gray-700 flex-shrink-0">
        <nav className="-mb-px flex space-x-6 overflow-x-auto" aria-label="Tabs">
          <button onClick={() => setActiveTab("all")} className={`whitespace-nowrap py-3 px-1 border-b-2 font-medium text-sm ${ activeTab === "all" ? "border-orange-500 text-orange-500" : "border-transparent text-gray-400 hover:text-gray-200 hover:border-gray-500" }`}>View all</button>
          {appCameraIds.map((appId, index) => (
            <button key={appId} onClick={() => setActiveTab(appId)} className={`whitespace-nowrap py-3 px-1 border-b-2 font-medium text-sm ${ activeTab === appId ? "border-orange-500 text-orange-500" : "border-transparent text-gray-400 hover:text-gray-200 hover:border-gray-500" }`}>
                {cameraNames[index]}
            </button>
          ))}
        </nav>
      </div>

      <div className="flex flex-grow min-h-0 gap-4">
        <div className="w-2/3 bg-gray-800 rounded-md p-1 flex items-center justify-center">
          {activeTab === "all" && (
            <div className="grid grid-cols-2 grid-rows-2 gap-1 w-full h-full">
              {appCameraIds.map((appId) => {
                const config = cameraFrameConfig[appId];
                const jsonCameraId = appCameraIdToJsonId[appId];
                const tracks = jsonCameraId ? currentFrameData?.cameras?.[jsonCameraId]?.tracks : null;
                return (
                  <ImageSequencePlayer
                    key={appId} cameraId={appId} basePath={config?.basePath || ""}
                    startFrame={config?.startFrame ?? 0}
                    frameCount={config?.frameCount || 1}
                    currentFrameIndex={currentGlobalFrameIndex}
                    imageExtension={config?.extension}
                    tracks={tracks || null} className="min-h-0"
                  />);
              })}
            </div>
          )}
          {appCameraIds.includes(activeTab) && (() => {
              const currentAppId = activeTab;
              const config = cameraFrameConfig[currentAppId];
              const jsonCameraId = appCameraIdToJsonId[currentAppId];
              const tracks = jsonCameraId ? currentFrameData?.cameras?.[jsonCameraId]?.tracks : null;
              return (
                <ImageSequencePlayer
                  key={currentAppId} cameraId={currentAppId} basePath={config?.basePath || ""}
                  startFrame={config?.startFrame ?? 0}
                  frameCount={config?.frameCount || 1}
                  currentFrameIndex={currentGlobalFrameIndex}
                  imageExtension={config?.extension}
                  tracks={tracks || null} className="w-full h-full"
                />);
            })()}
        </div>

        <div className="w-1/3 flex flex-col gap-4">
            <div ref={overallMapContainerRef} className="bg-gray-700 rounded-md h-1/2 grid grid-cols-2 grid-rows-2 gap-px overflow-hidden">
                {appCameraIds.map((appId, index) => {
                    const jsonCameraId = appCameraIdToJsonId[appId];
                    const pointsForThisQuadrant = perCameraMapPoints[jsonCameraId] || [];
                    const quadrantColor = cameraColorMap[jsonCameraId] || defaultDotColor;
                    const camName = cameraNames[index];
                    return (
                        <div key={`map-quad-${jsonCameraId}`} className="relative w-full h-full bg-gray-800 p-1" title={`Map - ${camName}`}>
                            <div className="absolute top-1 left-1 text-xs text-gray-400 opacity-75 pointer-events-none">{camName}</div>
                            {pointsForThisQuadrant.map((point) => (
                                <div
                                    key={`${jsonCameraId}-dot-${point.globalId}`}
                                    title={`ID: ${point.globalId} (from ${camName})`}
                                    className={`absolute w-2 h-2 ${quadrantColor} rounded-full shadow-md`}
                                    style={{ left: `${point.x}px`, top: `${point.y}px`, transform: 'translate(-50%, -50%)', pointerEvents: 'none' }}
                                />
                            ))}
                        </div>
                    );
                })}
                {overallMapDimensions.width === 0 && !currentFrameData && (
                    <div className="absolute inset-0 flex items-center justify-center text-gray-400 col-span-2 row-span-2">
                        {connectionError || (isConnecting ? "Connecting..." : "Awaiting data...")}
                    </div>
                )}
            </div>

            <div className="flex flex-grow gap-4 h-1/2">
                <div className="bg-gray-800 rounded-md p-4 w-1/2 flex flex-col">
                    <h3 className="text-sm font-semibold mb-3 text-gray-400">Detections per camera</h3>
                    <div className="space-y-2 text-xs flex-grow overflow-y-auto">
                        {appCameraIds.map((appId, index) => {
                            const jsonCamId = appCameraIdToJsonId[appId];
                            const numDetections = currentFrameData?.cameras?.[jsonCamId]?.tracks?.length || 0;
                            const maxDetAcrossAll = Math.max(1, ...appCameraIds.map(id => currentFrameData?.cameras?.[appCameraIdToJsonId[id]]?.tracks?.length || 0));
                            return (
                                <div key={`det-${cameraNames[index]}`} className="flex items-center justify-between">
                                    <span>{cameraNames[index]}</span>
                                    <div className="flex items-center">
                                        <div className="w-16 h-2 bg-gray-600 rounded-full mr-2">
                                            <div className="h-2 bg-green-500 rounded-full" style={{ width: `${maxDetAcrossAll > 0 ? (numDetections / maxDetAcrossAll) * 100 : 0}%` }}></div>
                                        </div>
                                        <span className="font-mono">{numDetections}</span>
                                    </div>
                                </div>
                            );
                        })}
                    </div>
                </div>
                <div className="bg-gray-800 rounded-md p-4 w-1/2 flex flex-col justify-between">
                    <div>
                        <div className="flex justify-between items-center mb-2">
                            <h3 className="text-sm font-semibold text-gray-400">Last caught</h3>
                            <span className="text-green-400 text-xs font-semibold flex items-center">
                                <span className="h-1.5 w-1.5 bg-green-400 rounded-full mr-1"></span>Active
                            </span>
                        </div>
                        <p className="text-xs">Camera <span className="font-bold text-lg">N/A</span></p>
                        <p className="text-xs mt-3">Person Id: <span className="font-bold">N/A</span></p>
                        <p className="text-xs mt-1">Tracking start: N/A</p>
                        <p className="text-xs mt-1">Last known: N/A</p>
                    </div>
                    <button className="w-full py-1.5 bg-red-600 hover:bg-red-700 text-white rounded text-sm font-semibold mt-3">Cancel Tracking</button>
                </div>
            </div>
        </div>
      </div>
    </div>
  );
};

export default GroupViewPage;
